### 🔶 Step 2: Score Candidates by Similarity to ψ⁰

```python
def score_candidate_against_kernel(feature, kernel):
    """
    Scores a candidate feature against ψ⁰ kernel based on normalized similarity.
    """
    # Pull values
    area = feature.get('area_sq_meters')
    height = feature.get('mean_dsm_elevation')
    aspect = feature.get('aspect_ratio')

    # Z-score style similarity
    area_score = ee.Number(area).subtract(kernel.get('area_mean')).abs().divide(kernel.get('area_sd'))
    height_score = ee.Number(height).subtract(kernel.get('height_mean')).abs().divide(kernel.get('height_sd'))
    aspect_score = ee.Number(aspect).subtract(kernel.get('aspect_mean')).abs().divide(kernel.get('aspect_sd'))

    # Combine: lower = better (distance to ψ⁰)
    combined = area_score.add(height_score).add(aspect_score)

    # Invert to form similarity score (higher = better)
    similarity = ee.Number(1).divide(combined.add(1))  # Add 1 to prevent div by zero

    return feature.set('RE_similarity_score', similarity)
```
